\begin{UseCase}{CU-17}{Consultar ETS inscritos}{
		Este caso de uso permite al alumno consultar los ETS que tiene asignados.
	}
	\UCitem{Versión}{\color{Gray}1.0}
	\UCitem{Autor}{\color{Gray}De la cruz De la cruz Alejandra}
	\UCitem{Supervisa}{\color{Gray}Huertas Ramírez Daniel Martín}
	\UCitem{Actor}{\hyperlink{Alumno}{Alumno}}
	\UCitem{Propósito}{Permitir al alumno consultar los ETS que ha inscrito.}
	\UCitem{Entradas}{Seleccionar un periodo}
	\UCitem{Origen}{Pantalla táctil}
	\UCitem{Salidas}{Lista de ETS inscritos.}
	\UCitem{Destino}{\IUref{IU15}{Pantalla Consultar ETS del alumno}}
	\UCitem{Precondiciones}{El alumno debe estar autenticado en el sistema.}
	\UCitem{Postcondiciones}{El alumno ha consultado los ETS inscritos.}
	\UCitem{Errores}{
			E1: El sistema no puede recuperar la información de los ETS asignados y muestra el mensaje {\bf MSG-26}{``No hay ETS inscritos actualmente.''}
			
			E2: No hay ETS inscritos {\bf MS-G9}{``Error al consultar la base de datos. Intente nuevamente más tarde.''}.
	}
	\UCitem{Tipo}{Se extiende del CU-01 Iniciar sesión del sistema móvil}
	\UCitem{Observaciones}{Ninguna}
\end{UseCase}
%--------------------------------------
\begin{UCtrayectoria}
	\UCpaso[\UCactor] El alumno selecciona el período académico que desea consultar desde la \IUref{IU14}{Pantalla Periodo de ETS del alumno}.
	\UCpaso El sistema verifica que el alumno tenga ETS inscritos en el periodo seleccionado \Trayref{A}.
	\UCpaso El sistema despliega la lista de ETS inscritos en la \IUref{IU15}{Pantalla Consultar ETS del alumno}.
\end{UCtrayectoria}

%--------------------------------------        
\begin{UCtrayectoriaA}{A}{No hay ETS inscritos en el periodo seleccionado}
	\UCpaso El sistema muestra un mensaje: {\bf MSG-26}{``No hay ETS inscritos actualmente.''}
	\UCpaso[\UCactor] El alumno presiona el botón \IUbutton{Regresar} para volver a la pantalla anterior.
	\UCpaso Fin de la trayectoria alternativa.
\end{UCtrayectoriaA}
%--------------------------------------        
\begin{UCtrayectoriaA}{B}{Error en la conexión con la base de datos}
	\UCpaso El sistema muestra un mensaje de error: {\bf MSG-09}{``Error al consultar la base de datos. Intente nuevamente más tarde.''}
	\UCpaso[\UCactor] El alumno presiona el botón \IUbutton{Aceptar} para cerrar el mensaje.
	\UCpaso[\UCactor] El alumno puede intentar la consulta nuevamente o presionar el botón \IUbutton{Regresar} para volver a la pantalla anterior.
	\UCpaso Fin de la trayectoria alternativa.
\end{UCtrayectoriaA}
